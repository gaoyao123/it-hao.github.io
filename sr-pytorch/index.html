def is_image_file(filename, mode=[".png", ".jpg", ".jpeg", ".bmp"]):
    """
    判断以 mode 中的任一后缀名的文件是否存在
    """
    return any(filename.endswith(extension) for extension in mode)

def filenames_from_dir(root_dir, filenames, mode = [".png", ".jpg", ".jpeg", ".bmp"]):
    """mode: 图像格式
    root_dir (string):图像集的根目录
    filenames:保存文件名的列表
    """
    for filename in os.listdir(root_dir):
        filepath = os.path.join(root_dir, filename)
        # 如果是图片
        if is_image_file(filepath):
            filenames.append(filepath)
        # 如果是文件夹
        elif os.path.isdir(filepath):
            filenames_from_dir(filepath,filenames,mode)

def get_filePath_Name_Ext(filename):
    (filepath, tempfilename) = os.path.split(filename)
    (shotname,extension) = os.path.splitext(tempfilename)
    return filepath,shotname,extension

def crop_image_set(root_dir, height, width, stride):
    """crop images into a database with window(height,width,stride)
    切割图片，进一步拓展数据集
    """
    save_path=root_dir + '\\crops'
    if not os.path.exists(save_path):
        os.makedirs(save_path)
    file_names = []
    # 得到所有图片的绝对路径，包括子文件夹中的图片
    filenames_from_dir(root_dir, file_names)
    cropLen = 0
    for i in file_names:
        im = Image.open(i)
        # 分别获得文件夹的根目录，图片名，图片的后缀名
        root_dir, filename, exten = get_filePath_Name_Ext(i)
        if((im.size[0] > width) and (im.size[1] > height)):
            for x in range(0, im.size[0]-width, stride):
                for y in range(0, im.size[1]-height, stride):
                    crop_dir = save_path+'\\'+filename+'cropxy_'+str(x)+'_'+str(y)+exten
                    region = im.crop((x,y,x+width,y+height))
                    region.save(crop_dir)
                    cropLen +=1
    return cropLen

class SRdataset(Dataset):
    """从文件夹中加载图像；但是需要先进行图片的切块，图像预处理；加载图像集
    """
    def __init__(self, root_dir, height=128, width=128, stride=96, downscale=3, is_gray = False, transform=None,
                 rotate = False, rescale = False):
        """root_dir (string):图像集的根目录
        transform (callable, optional):可选择进行的图像变换
        channel: 1 CSV, 3 RGB
        """
        self.transform = transform        
        self.is_gray = is_gray
        self.datalen = crop_image_set(root_dir,height, width, stride)
        self.filenames = []
        filenames_from_dir(root_dir+'\\crops', self.filenames)
        self.downscale = downscale
        self.rotate = rotate
        self.rescale = rescale
        self.totensor = transforms.ToTensor()
    def __len__(self):
        return self.datalen

    def __getitem__(self, idx):
        lab = Image.open(self.filenames[idx]).convert('RGB')

        if self.rotate:
            dv = random.randint(1,3)
            lab = lab.rotate(90*dv, expand = True)
        if self.rescale:
            dv = random.randint(8,9)
            lab = lab.resize((lab.size[0]*dv // 10, lab.size[1]*dv // 10))
        if self.transform:
            lab = self.transform(lab)

        size1 = int(lab.size[1] / self.downscale)
        size0 = int(lab.size[0] / self.downscale)

        self.downtrans = transforms.Compose([
            transforms.Scale((size1, size0), interpolation=Image.BICUBIC),
            transforms.Scale((lab.size[1], lab.size[0]), interpolation=Image.BICUBIC)])
        if self.is_gray:
            lab = lab.convert('YCbCr')
        # 分别返回 data 和 label
        data = self.downtrans(lab)
        return {'image': self.totensor(data),
                'label': self.totensor(lab)}

class SRdataset3(Dataset):
    """从文件夹中加载图像；图像预处理；加载图像集
    """
    def __init__(self, root_dir, downscale=4, is_gray = False, transform=None,
                 rotate = False, rescale = False, lr_sr = True):
        """root_dir (string):图像集的根目录
        transform (callable, optional):可选择进行的图像变换
        channel: 1 CSV, 3 RGB
        """
        self.transform = transform        
        self.is_gray = is_gray
        self.filenames = []
        filenames_from_dir(root_dir, self.filenames)
        self.downscale = downscale
        self.rotate = rotate
        self.totensor = transforms.ToTensor()
        self.lr_sr = lr_sr
    def __len__(self):
        return len(self.filenames)

    def __getitem__(self, idx):
        lab = Image.open(self.filenames[idx]).convert('RGB')

        if self.rotate:
            dv = random.randint(1,3)
            lab = lab.rotate(90*dv, expand = True) # 旋转对象返回一个新对象，（旋转角度，expand设置为True→调整整个尺寸以适应旋转后的新图像）

        if self.transform:
            lab = self.transform(lab)

        # 缩小训练的图片尺寸
        size1 = int(lab.size[1] / self.downscale)
        size0 = int(lab.size[0] / self.downscale)
        # 标签图片的尺寸不变
        lab_size0 = int(size0 * self.downscale)
        lab_size1 = int(size1 * self.downscale)
        lab = lab.crop((0, 0, lab_size0, lab_size1))

        # 判断此处的图片的像素是不是过大，过大的话需要对其进行降采样
        if self.lr_sr:
            self.downtrans = transforms.Scale((size1, size0), interpolation=Image.BICUBIC)
        else:
            self.downtrans = transforms.Compose([
                transforms.Scale((size1, size0), interpolation=Image.BICUBIC),
                transforms.Scale((lab.size[1], lab.size[0]), interpolation=Image.BICUBIC)])

        if self.is_gray:
            lab = lab.convert('YCbCr')
        data = self.downtrans(lab)
        return {'image': self.totensor(data),
                'label': self.totensor(lab)}


class Modcrop(object):
    """按照给定步长去除图像边缘余数，更改一个图像的尺寸
    Args:
        window: 滑动窗口
        stride: 步长。 如果是个元组类型, step(0):横向；step(1)：纵向。
        如果int类型,横向与纵向的步长一致。
    """
    def __init__(self, window, stride):
        assert isinstance(window, (int, tuple))
        assert isinstance(stride, (int, tuple))
        self.window = window
        self.stride = stride

    def __call__(self, image):
        w, h = image.size[0], image.size[1]
        if isinstance(self.window, int):
            h_ = h - self.window
            w_ = w - self.window
        else:
            new_h, new_w = self.window
            h_ = h - new_h
            w_ = w - new_w

        if isinstance(self.stride, int):
            s_h = self.stride
            s_w = self.stride
        else:
            s_h, s_w = self.stride

        h = h - np.mod(h_, s_h)
        w = w - np.mod(w_, s_w)
        return image.resize((w,h))
