在做SR实验中，因为网络深度，显存较小等问题，必须对图片进行处理，以便能够得到更好的实验结果。

SR数据处理

1、图片多尺度处理

这里主要的思想是：

  • 将尺度大于 512×512 的图片按照0.5.->06->···1.0的尺度分别保存到单独的文件夹中；

  • 将尺度小于 512×512 的图片单独保存在一个文件夹中；

代码如下：

def classfily_image(image_dir):
    """
    按照图片大小将其分类到不同尺度的文件夹中
    """
    file_names = []
    # 得到所有图片的绝对路径
    filenames_from_dir(image_dir, file_names)
    for i in file_names:
        image = Image.open(i)
        # 分别获得文件夹的根目录，图片名，图片的后缀名
        image_dir, filename, exten = get_filePath_Name_Ext(i)
        # crop_dir1 = save_path+'\\'+filename+'cropxy_'+str(x)+'_'+str(y)+exten
        image_size = min(image.size[0],image.size[1])
        # image_size < 512的单独进行处理，然后image_size > 512 的将他们的尺度分别缩小为0.5~1倍
        if image_size < 512:
            image_save_dir = image_dir + "\\small_size"
            create_new_folder(image_save_dir)
            image_save_name = image_save_dir + "\\" + filename + "_small_size" + exten
            # save image
            image.save(image_save_name)
        else:
            # scale = round(512 / image_size, 1) + 0.1 # 这里不能四舍五入
            scale = (int(5120/image_size))/10 + 0.1
            # 图片的大小是 >512  and  <1024
            if scale <= 1:
                # for x in np.arange(max(scale,0.5), 1, 0.1): # 容易出现问题
                for x in np.arange(max(scale,0.5),1.01,0.1):
                    downtrans = transforms.Resize((int(round(image.size[1]*x)),int(round(image.size[0]*x))),interpolation=Image.BICUBIC)
                    new_image = downtrans(image)
                    image_save_dir = image_dir + "\\" + str(x) +"_size"
                    create_new_folder(image_save_dir)
                    image_save_name = image_save_dir + "\\" + filename +"_"+ str(x)+ "_size" + exten
                    # save
                    new_image.save(image_save_name)

2、图片切块

这里主要是采用了两种方法：

  • 五分法：将所有图片切成5块，主要想法是切成 512×512 大小，在论文中，4倍下采样为 128×128 ，通过算法恢复成为 原始大小；也可以切成 384×384 大小，3倍下采样为 128×128 ；或者切成 256×256 大小，2倍下采样为 128×128。

  • 滑窗法：这里采用的就是 stride 来进行一定的滑动，以便获得更多的图片，但是在之前实验中表明这样的效果不是很好，但是为了得到一整图片的整体特征的话，这种方法得到的图片还是有用的，但是这样的label不需要很多。

def crop_fiveImages(image_dir,width,height):
    """
    五分法：
    """
    image_save_dir = image_dir + '\\edges-' + str(width) # 保存边缘图片
    image_save_dir2 = image_dir + '\\centers-' + str(width) # 保存中间图片
    create_new_folder(image_save_dir)
    create_new_folder(image_save_dir2)
    file_names = []
    cropLen = 0
    filenames_from_dir(image_dir, file_names)
    # 得到所有图片的绝对路径
    for i in file_names:
        img = Image.open(i)
        root_dir, filename, exten = get_filePath_Name_Ext(i)
        size1 = img.size[1] # Height
        size0 = img.size[0] # Width
        for x in [0,img.size[0]-width]:
            for y in [0,img.size[1]-height]:
                crop_dir1 = image_save_dir + '\\' + filename + 'cropxy_' + str(x) + '_' + str(y) + exten
                region = img.crop((x,y,x+width,y+height))
                region.save(crop_dir1)
                cropLen += 1
        x = img.size[0]//2 - width//2
        y = img.size[1]//2 - height//2
        crop_dir2 = image_save_dir2+'\\'+filename+'cropxy_'+str(x)+'_'+str(y)+exten
        region = img.crop((x,y,x+width,y+height))
        region.save(crop_dir2)
        cropLen += 1
    return cropLen

def crop_slideWindows(image_dir,width,height,stride):
    """
    滑窗法：
    """
    save_path=image_dir + '\\crops'
    create_new_folder(save_path)
    file_names = []
    # 得到所有图片的绝对路径，包括子文件夹中的图片
    filenames_from_dir(image_dir, file_names)
    cropLen = 0
    for i in file_names:
        im = Image.open(i)
        # 分别获得文件夹的根目录，图片名，图片的后缀名
        root_dir, filename, exten = get_filePath_Name_Ext(i)
        if((im.size[0] > width) and (im.size[1] > height)):
            for x in range(0, im.size[0]-width, stride):
                for y in range(0, im.size[1]-height, stride):
                    crop_dir = save_path+'\\'+filename+'cropxy_'+str(x)+'_'+str(y)+exten
                    region = im.crop((x,y,x+width,y+height))
                    region.save(crop_dir)
                    cropLen +=1
    return cropLen
